server.mjs — 服务端 Node.js 
// ===== 服务端：基本 SSE 端点 =====
if (pathname === '/events') {
  // 设置 SSE 必需的响应头
  res.writeHead(200, {
    'Content-Type': 'text/event-stream',
    'Cache-Control': 'no-cache',
    'Connection': 'keep-alive',
    'Access-Control-Allow-Origin': '*',
  })

  let count = 0
  const intervalId = setInterval(() => {
    count++
    // SSE 格式: data: <json>\n\n
    res.write(`id: ${count}\ndata: ${JSON.stringify({
      id: count,
      message: `第 ${count} 条推送`,
      timestamp: new Date().toISOString()
    })}\n\n`)

    if (count >= 10) {
      clearInterval(intervalId)
      res.write(`event: done\ndata: 全部推送完成\n\n`)
      res.end()
    }
  }, 1000) // 每秒推一条

  // 客户端断开时清理资源
  req.on('close', () => clearInterval(intervalId))
}
index.html — 客户端 JavaScript 
// ===== 客户端：原生 EventSource API =====
sseSource = new EventSource('/events')

// 连接成功回调
sseSource.onopen = () => {
  console.log('SSE 连接已建立')
}

// 监听自定义事件（服务端 event: connected）
sseSource.addEventListener('connected', (e) => {
  console.log('连接成功:', e.data)
})

// 接收普通消息（没有 event 字段的消息）
sseSource.onmessage = (e) => {
  const data = JSON.parse(e.data)
  console.log('收到消息:', data.message)
}

// 监听 done 事件（服务端 event: done）
sseSource.addEventListener('done', (e) => {
  console.log('推送完成:', e.data)
  sseSource.close() // 手动关闭连接
})

// 错误处理（浏览器会自动重连）
sseSource.onerror = (err) => {
  console.error('连接异常，将自动重连')
}

// 手动关闭
function stopSSE() {
  sseSource.close()
}

server.mjs — 服务端：逐字推送 Node.js 
// ===== 服务端：模拟 AI 逐字输出 =====
if (pathname === '/chat') {
  const text = url.searchParams.get('text')
  // ... 设置 SSE 响应头 ...

  let index = 0
  const speed = 80 // 每字间隔 80ms

  const intervalId = setInterval(() => {
    if (index < text.length) {
      // 逐字推送，每次只推一个字符
      const char = text[index]
      res.write(`data: ${JSON.stringify({
        token: char,   // 单个字符
        index          // 位置序号
      })}\n\n`)
      index++
    } else {
      clearInterval(intervalId)
      // [DONE] 是 OpenAI API 的结束标志
      res.write(`event: done\ndata: [DONE]\n\n`)
      res.end()
    }
  }, speed)
}
index.html — 客户端：fetch 解析 SSE 流 JavaScript 
// ===== 客户端：fetch + ReadableStream 解析 =====
const response = await fetch(`/chat?text=${encodeURIComponent(text)}`)

// 获取流的读取器
const reader = response.body.getReader()
const decoder = new TextDecoder()
let buffer = ''

while (true) {
  const { done, value } = await reader.read()
  if (done) break

  // 把二进制解码为文本，追加到 buffer
  buffer += decoder.decode(value, { stream: true })

  // SSE 协议：消息以 \n\n 分隔
  const lines = buffer.split('\n\n')
  buffer = lines.pop() || '' // 最后一个可能不完整

  for (const line of lines) {
    if (line.startsWith('data: ')) {
      const dataStr = line.slice(6)
      if (dataStr === '[DONE]') continue

      const data = JSON.parse(dataStr)
      if (data.token) {
        // 逐字渲染到页面，实现打字机效果
        contentSpan.appendChild(
          document.createTextNode(data.token)
        )
      }
    }
  }
}

server.mjs — 服务端：读取自定义请求头 Node.js 
// ===== 服务端：从请求头中读取自定义参数 =====
if (pathname === '/fetch-sse') {
  // 从请求头中读取鉴权信息
  const auth = req.headers['authorization'] || 'no-auth'
  const customHeader = req.headers['x-custom-header'] || 'none'

  console.log(`收到自定义请求头: ${auth}, ${customHeader}`)

  // ... 设置 SSE 响应头 ...

  // 先发一条确认消息，告诉客户端自定义头已收到
  res.write(`event: auth\ndata: ${JSON.stringify({
    auth, customHeader,
    message: '自定义请求头已收到！'
  })}\n\n`)

  // 然后正常推送消息...
  const messages = [
    '使用 fetch API 也能实现 SSE！',
    '这样可以自定义请求头。',
    '比如传递 Token 进行鉴权。',
    '灵活性比 EventSource 高很多！',
  ]
  messages.forEach((msg, i) => {
    setTimeout(() => {
      res.write(`data: ${JSON.stringify({ index: i+1, message: msg })}\n\n`)
      if (i === messages.length - 1) {
        setTimeout(() => {
          res.write(`event: done\ndata: 全部完成\n\n`)
          res.end()
        }, 300)
      }
    }, (i + 1) * 800)
  })
}
index.html — 客户端：fetch 携带自定义请求头 JavaScript 
// ===== 客户端：fetch 携带自定义请求头 =====
const response = await fetch('/fetch-sse', {
  headers: {
    'Authorization': 'Bearer demo-token-12345',
    'X-Custom-Header': 'sugar-sse-demo',
    'Accept': 'text/event-stream',  // SSE 专用 Accept
  },
})

// 拿到 body 的 ReadableStream 读取器
const reader = response.body.getReader()
const decoder = new TextDecoder()
let buffer = ''

while (true) {
  const { done, value } = await reader.read()
  if (done) break

  buffer += decoder.decode(value, { stream: true })
  const lines = buffer.split('\n\n')
  buffer = lines.pop() || ''

  for (const line of lines) {
    // 用正则提取 event 和 data 字段
    const dataMatch = line.match(/^data: (.+)$/m)
    const eventMatch = line.match(/^event: (.+)$/m)

    if (dataMatch) {
      const data = JSON.parse(dataMatch[1])
      console.log(`[${eventMatch?.[1] || 'data'}]`, data.message)
    }
  }
}
💡 笔记：
① EventSource API 是原生 SSE 方案，简单易用，但只支持 GET，不能带自定义请求头。
② Cookie 跨域可以带 ✅ — 使用 new EventSource(url, { withCredentials: true })，服务端需设 Access-Control-Allow-Credentials: true 且 Origin 不能为 *。
③ fetch + ReadableStream 可以突破 EventSource 的限制，支持 POST、自定义头（如 Authorization）、鉴权等。
④ LLM 流式输出 的核心就是 SSE 或 fetch SSE，每次 push 一个 token，前端逐字渲染。
⑤ SSE 连接在 HTTP/1.1 下同域名最多 6 个，HTTP/2 下最多 100 个。
⑥ 服务端记得设 Cache-Control: no-cache，代理服务器要关缓冲 proxy_buffering off。
